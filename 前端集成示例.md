# 智忆助理 - 前端集成示例

## WebSocket 连接（接收实时提醒）

```javascript
// 建立 WebSocket 连接
const userId = 'user123';
const ws = new WebSocket(`ws://localhost:8000/ws/${userId}`);

// 连接建立
ws.onopen = () => {
    console.log('WebSocket 已连接');
    // 发送心跳（保持连接）
    setInterval(() => {
        if (ws.readyState === WebSocket.OPEN) {
            ws.send('ping');
        }
    }, 30000);
};

// 接收消息
ws.onmessage = (event) => {
    const data = JSON.parse(event.data);

    if (data.type === 'reminder') {
        // 收到提醒！
        showNotification(data.title, data.content);
        playNotificationSound();
    } else if (data.type === 'pong') {
        // 心跳响应
        console.log('服务器心跳:', data.timestamp);
    }
};

// 显示浏览器通知
function showNotification(title, content) {
    if (Notification.permission === 'granted') {
        new Notification(title, {
            body: content,
            icon: '/notification-icon.png'
        });
    } else {
        // 使用自定义弹窗
        alert(`⏰ ${title}\n${content}`);
    }
}

// 播放提醒音
function playNotificationSound() {
    const audio = new Audio('/notification-sound.mp3');
    audio.play().catch(e => console.log('播放提醒音失败:', e));
}

// 请求通知权限
if ('Notification' in window && Notification.permission === 'default') {
    Notification.requestPermission();
}
```

## 创建提醒（API 调用）

```javascript
// 创建提醒
async function createReminder(content, time) {
    const response = await fetch('/api/tasks', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            user_id: userId,
            content: content,
            reminder_time: time,  // 支持自然语言如 "明天下午3点"
            title: '自定义提醒'
        })
    });

    const result = await response.json();
    if (result.success) {
        console.log('提醒已设置:', result.message);
        return result.task_id;
    } else {
        console.error('设置提醒失败:', result.error);
    }
}

// 使用示例
createReminder('记得看论文', '明天下午3点');
createReminder('喝水', '2026-03-13T15:00:00');  // ISO格式也支持
```

## 取消任务

```javascript
async function cancelTask(taskId) {
    const response = await fetch(`/api/tasks/${taskId}?user_id=${userId}`, {
        method: 'DELETE'
    });

    const result = await response.json();
    if (result.success) {
        console.log('任务已取消');
    }
}
```

## 聊天接口（已集成提醒功能）

```javascript
// 发送消息 - 系统会自动识别提醒意图
async function sendMessage(message) {
    const response = await fetch('/api/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            user_id: userId,
            message: message,
            stream: false
        })
    });

    const result = await response.json();
    return result.data;
}

// 示例：设置提醒
// 系统会识别 EXECUTE_TASK 意图并自动创建定时任务
sendMessage('明天下午3点提醒我看论文');
// 回复：已为您设置提醒📅 时间：明天下午3点 内容：看论文 ...

sendMessage('每天早上9点提醒我吃维生素');
// 回复：已为您设置周期性提醒：每天早上9点 内容：吃维生素 ...
```

## Vue3 组件示例

```vue
<template>
  <div class="chat-container">
    <div class="messages">
      <div v-for="msg in messages" :key="msg.id" :class="['message', msg.role]">
        {{ msg.content }}
      </div>
    </div>

    <div class="input-area">
      <input v-model="inputMessage" @keyup.enter="sendMessage" placeholder="输入消息..." />
      <button @click="sendMessage">发送</button>
    </div>

    <!-- 提醒通知弹窗 -->
    <div v-if="currentReminder" class="reminder-popup">
      <h3>⏰ {{ currentReminder.title }}</h3>
      <p>{{ currentReminder.content }}</p>
      <button @click="dismissReminder">知道了</button>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue';

const userId = 'user123';
const messages = ref([]);
const inputMessage = ref('');
const currentReminder = ref(null);
let ws = null;

// WebSocket 连接
onMounted(() => {
  ws = new WebSocket(`ws://localhost:8000/ws/${userId}`);

  ws.onmessage = (event) => {
    const data = JSON.parse(event.data);

    if (data.type === 'reminder') {
      currentReminder.value = data;
      // 播放提醒音
      new Audio('/notification-sound.mp3').play();
    }
  };

  // 心跳
  const heartbeat = setInterval(() => {
    if (ws.readyState === WebSocket.OPEN) {
      ws.send('ping');
    }
  }, 30000);

  onUnmounted(() => {
    clearInterval(heartbeat);
    ws.close();
  });
});

// 发送消息
async function sendMessage() {
  if (!inputMessage.value.trim()) return;

  const message = inputMessage.value;
  messages.value.push({ id: Date.now(), role: 'user', content: message });
  inputMessage.value = '';

  const response = await fetch('/api/chat', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ user_id: userId, message })
  });

  const result = await response.json();
  messages.value.push({
    id: Date.now() + 1,
    role: 'assistant',
    content: result.data
  });
}

function dismissReminder() {
  currentReminder.value = null;
}
</script>
```

## 完整工作流程

```
用户: "明天下午3点提醒我看论文"
    ↓
后端: 识别 EXECUTE_TASK 意图
    ↓
后端: 解析时间 → 2026-03-13 15:00
    ↓
后端: 创建 ScheduledTask
    - 注册到 APScheduler
    - 保存到数据库
    - 存储到记忆库
    ↓
后端: 回复 "已为您设置提醒📅..."
    ↓
(时间到达)
    ↓
APScheduler: 触发任务
    ↓
后端: 通过 WebSocket 推送通知
    ↓
前端: 显示提醒弹窗 + 播放声音
```
